Para a realização dessa atividade eu utilizei os dados sobre os gastos dos parlamentares disponíveis no site de transparência do congresso nacional do ano de 2015. Utilizando apenas os gastos em COMBUSTÍVEIS E LUBRIFICANTES dos nossos parlamantares obtemos as seguintes informações:

# Dados da câmara dos deputados

Agora irei selecionar 10 amostras de tamanho 100 (escolhidos aleatoriamente) para verificar o valor da média dessas amostras. Para cada amostra irei verificar a média. O resultado das 100 médias estão mostradas no gráfico abaixo:

mediaAmostra <- function(original, samplesSize, numSamples) {
  dist_original <- original
  sample_size <- samplesSize
  num_samples <- numSamples
  
  samples_means <- c()
  for(i in seq(1, num_samples)){
    a_sample <- sample(dist_original, sample_size)
    samples_means[i] <- mean(a_sample)
    }
  samples_means
  }

samplesMeans <- mediaAmostra(vlrCombustiveis$vlrLiquido, 100, 10)

ggplot(data.frame(samplesMeans), aes(samplesMeans)) +
  geom_histogram() +
  theme_bw() + 
  labs(x='Média das amostras', y='Ocorrência') + ggtitle("10 Amostras")

multiplot <- function(..., plotlist=NULL, file, cols=1, layout=NULL) {
  require(grid)

  # Make a list from the ... arguments and plotlist
  plots <- c(list(...), plotlist)

  numPlots = length(plots)

  # If layout is NULL, then use 'cols' to determine layout
  if (is.null(layout)) {
    # Make the panel
    # ncol: Number of columns of plots
    # nrow: Number of rows needed, calculated from # of cols
    layout <- matrix(seq(1, cols * ceiling(numPlots/cols)),
                    ncol = cols, nrow = ceiling(numPlots/cols))
  }

 if (numPlots==1) {
    print(plots[[1]])

  } else {
    # Set up the page
    grid.newpage()
    pushViewport(viewport(layout = grid.layout(nrow(layout), ncol(layout))))

    # Make each plot, in the correct location
    for (i in 1:numPlots) {
      # Get the i,j matrix positions of the regions that contain this subplot
      matchidx <- as.data.frame(which(layout == i, arr.ind = TRUE))

      print(plots[[i]], vp = viewport(layout.pos.row = matchidx$row,
                                      layout.pos.col = matchidx$col))
    }
  }
}
